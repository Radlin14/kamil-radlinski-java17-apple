SELECT ISSUES.SUMMARY , ISSUESLISTS.NAME
FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT I.SUMMARY , U.FIRSTNAME , U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGENDTO = U.ID;

SELECT U.FIRSTNAME , U.LASTNAME , COUNT(*) AS TASKS_QUANTITY
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGENDTO
GROUP BY LASTNAME , FIRSTNAME
HAVING COUNT(*) > 1 ;
